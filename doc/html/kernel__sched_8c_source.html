16 /*
17  The thread layout.
18  --------------------
19 
20  On the x86 (Pentium) architecture, the stack grows upward. Therefore, we
21  can allocate the TCB at the top of the memory block used as the stack.
22 
23  +-------------+
24  | TCB |
25  +-------------+
26  | |
27  | stack |
28  | |
29  | ^ |
30  | | |
31  +-------------+
32  | first frame |
33  +-------------+
34 
35  Advantages: (a) unified memory area for stack and TCB (b) stack overrun will
36  crash own thread, before it affects other threads (which may make debugging
37  easier).
38 
39  Disadvantages: The stack cannot grow unless we move the whole TCB. Of course,
40  we do not support stack growth anyway!
41  */
42 
43 /*
44  A counter for active threads. By "active", we mean 'existing',
45  with the exception of idle threads (they don't count).
46  */

48 Mutex active_threads_spinlock = MUTEX_INIT;

112 TCB* spawn_thread(PCB* pcb, void (*func)())
113 {
114  /* The allocated thread size must be a multiple of page size */
115  TCB* tcb = (TCB*)allocate_thread(THREAD_SIZE);
116 
117  /* Set the owner */
118  tcb->owner_pcb = pcb;
119 
120  /* Initialize the other attributes */
121  tcb->type = NORMAL_THREAD;
122  tcb->state = INIT;
123  tcb->phase = CTX_CLEAN;
124  tcb->thread_func = func;
125  tcb->wakeup_time = NO_TIMEOUT;
126  rlnode_init(&tcb->sched_node, tcb); /* Intrusive list node */
127 
128  tcb->its = QUANTUM;
129  tcb->rts = QUANTUM;
130  tcb->last_cause = SCHED_IDLE;
131  tcb->curr_cause = SCHED_IDLE;
132 
133  /* Compute the stack segment address and size */
134  void* sp = ((void*)tcb) + THREAD_TCB_SIZE;
135 
136  /* Init the context */
137  cpu_initialize_context(&tcb->context, sp, THREAD_STACK_SIZE, thread_start);
138 
139 #ifndef NVALGRIND
140  tcb->valgrind_stack_id = VALGRIND_STACK_REGISTER(sp, sp + THREAD_STACK_SIZE);
141 #endif
142 
143  /* increase the count of active threads */
144  Mutex_Lock(&active_threads_spinlock);
145  active_threads++;
146  Mutex_Unlock(&active_threads_spinlock);
147 
148  return tcb;

203 /*
204  Possibly add TCB to the scheduler timeout list.
205 
206  *** MUST BE CALLED WITH sched_spinlock HELD ***
207 */
208 static void sched_register_timeout(TCB* tcb, TimerDuration timeout)
209 {
210  if (timeout != NO_TIMEOUT) {
211  /* set the wakeup time */
212  TimerDuration curtime = bios_clock();
213  tcb->wakeup_time = (timeout == NO_TIMEOUT) ? NO_TIMEOUT : curtime + timeout;
214 
215  /* add to the TIMEOUT_LIST in sorted order */
216  rlnode* n = TIMEOUT_LIST.next;
217  for (; n != &TIMEOUT_LIST; n = n->next)
218  /* skip earlier entries */
219  if (tcb->wakeup_time < n->tcb->wakeup_time)
220  break;
221  /* insert before n */
222  rl_splice(n->prev, &tcb->sched_node);
223  }
224 }

240 /*
241  Adjust the state of a thread to make it READY.
242 
243  *** MUST BE CALLED WITH sched_spinlock HELD ***
244  */
245 static void sched_make_ready(TCB* tcb)
246 {
247  assert(tcb->state == STOPPED || tcb->state == INIT);
248 
249  /* Possibly remove from TIMEOUT_LIST */
250  if (tcb->wakeup_time != NO_TIMEOUT) {
251  /* tcb is in TIMEOUT_LIST, fix it */
252  assert(tcb->sched_node.next != &(tcb->sched_node) && tcb->state == STOPPED);
253  rlist_remove(&tcb->sched_node);
254  tcb->wakeup_time = NO_TIMEOUT;
255  }
256 
257  /* Mark as ready */
258  tcb->state = READY;
259 
260  /* Possibly add to the scheduler queue */
261  if (tcb->phase == CTX_CLEAN)
262  sched_queue_add(tcb);
263 }

265 /*
266  Scan the \c TIMEOUT_LIST for threads whose timeout has expired, and
267  wake them up.
268 
269  *** MUST BE CALLED WITH sched_spinlock HELD ***
270 */
271 static void sched_wakeup_expired_timeouts()
272 {
273  /* Empty the timeout list up to the current time and wake up each thread */
274  TimerDuration curtime = bios_clock();
275 
276  while (!is_rlist_empty(&TIMEOUT_LIST)) {
277  TCB* tcb = TIMEOUT_LIST.next->tcb;
278  if (tcb->wakeup_time > curtime)
279  break;
280  sched_make_ready(tcb);
281  }
282 }

284 /*
285  Remove the head of the scheduler list, if any, and
286  return it. Return NULL if the list is empty.
287 
288  *** MUST BE CALLED WITH sched_spinlock HELD ***
289 */
290 static TCB* sched_queue_select(TCB* current)
291 {
292  /* Get the head of the SCHED list */
293  rlnode * sel = rlist_pop_front(SCHED[current->priority]);
294 
295  TCB* next_thread = sel->tcb; /* When the list is empty, this is NULL */
296 
297  if (next_thread == NULL)
298  next_thread = (current->state == READY) ? current : &CURCORE.idle_thread;
299 
300  next_thread->its = QUANTUM;
301 
302  return next_thread;
303 }

305 /*
306  Make the process ready.
307  */
308 int wakeup(TCB* tcb)
309 {
310  int ret = 0;
311 
312  /* Preemption off */
313  int oldpre = preempt_off;
314 
315  /* To touch tcb->state, we must get the spinlock. */
316  Mutex_Lock(&sched_spinlock);
317 
318  if (tcb->state == STOPPED || tcb->state == INIT) {
319  sched_make_ready(tcb);
320  ret = 1;
321  }
322 
323  Mutex_Unlock(&sched_spinlock);
324 
325  /* Restore preemption state */
326  if (oldpre)
327  preempt_on;
328 
329  return ret;
330 }

332 /*
333  Atomically put the current process to sleep, after unlocking mx.
334  */
335 void sleep_releasing(Thread_state state, Mutex* mx, enum SCHED_CAUSE cause,
336  TimerDuration timeout)
337 {
338  assert(state == STOPPED || state == EXITED);
339 
340  TCB* tcb = CURTHREAD;
341 
342  int preempt = preempt_off;
343  Mutex_Lock(&sched_spinlock);
344 
345  /* mark the thread as stopped or exited */
346  tcb->state = state;
347 
348  /* register the timeout (if any) for the sleeping thread */
349  if (state != EXITED)
350  sched_register_timeout(tcb, timeout);
351 
352  /* Release mx */
353  if (mx != NULL)
354  Mutex_Unlock(mx);
355 
356  /* Release the schduler spinlock before calling yield() !!! */
357  Mutex_Unlock(&sched_spinlock);
358 
359  /* call this to schedule someone else */
360  yield(cause);
361 
362  /* Restore preemption state */
363  if (preempt)
364  preempt_on;
365 }

369 void yield(enum SCHED_CAUSE cause)
370 {
371  /* Reset the timer, so that we are not interrupted by ALARM */
372  TimerDuration remaining = bios_cancel_timer();
373 
374  /* We must stop preemption but save it! */
375  int preempt = preempt_off;
376 
377  TCB* current = CURTHREAD; /* Make a local copy of current process, for speed */
378 
379  Mutex_Lock(&sched_spinlock);
380 
381  /* Update CURTHREAD state */
382  if (current->state == RUNNING)
383  current->state = READY;
384 
385  /* Update CURTHREAD scheduler data */
386  current->rts = remaining;
387  current->last_cause = current->curr_cause;
388  current->curr_cause = cause;
389 
390  /* Wake up threads whose sleep timeout has expired */
391  sched_wakeup_expired_timeouts();
392 
393  /* Get next */
394  TCB* next = sched_queue_select(current);
395  assert(next != NULL);
396 
397  /* Save the current TCB for the gain phase */
398  CURCORE.previous_thread = current;
399 
400  Mutex_Unlock(&sched_spinlock);
401 
402  /* Switch contexts */
403  if (current != next) {
404  CURTHREAD = next;
405  cpu_swap_context(&current->context, &next->context);
406  }
407 
408  /* This is where we get after we are switched back on! A long time
409  may have passed. Start a new timeslice...
410  */
411  gain(preempt);
412 }
413 
414 /*
415  This function must be called at the beginning of each new timeslice.
416  This is done mostly from inside yield().
417  However, for threads that are executed for the first time, this
418  has to happen in thread_start.
419 
420  The 'preempt' argument determines whether preemption is turned on
421  in the new timeslice. When returning to threads in the non-preemptive
422  domain (e.g., waiting at some driver), we need to not turn preemption
423  on!
424 */
425 
426 void gain(int preempt)
427 {
428  Mutex_Lock(&sched_spinlock);
429 
430  TCB* current = CURTHREAD;
431 
432  /* Mark current state */
433  current->state = RUNNING;
434  current->phase = CTX_DIRTY;
435  current->rts = current->its;
436 
437  /* Take care of the previous thread */
438  TCB* prev = CURCORE.previous_thread;
439  if (current != prev) {
440  prev->phase = CTX_CLEAN;
441  switch (prev->state) {
442  case READY:
443  if (prev->type != IDLE_THREAD)
444  sched_queue_add(prev);
445  break;
446  case EXITED:
447  release_TCB(prev);
448  break;
449  case STOPPED:
450  break;
451  default:
452  assert(0); /* prev->state should not be INIT or RUNNING ! */
453  }
454  }
455 
456  Mutex_Unlock(&sched_spinlock);
457 
458  /* Reset preemption as needed */
459  if (preempt)
460  preempt_on;
461 
462  /* Set a 1-quantum alarm */
463  bios_set_timer(current->rts);
464 }

467 {
