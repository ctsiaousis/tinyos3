95 {

115 int ParseProcInfo(procinfo* pinfo, Program* prog, int argc, const char** argv )
116 {
117  if(pinfo->main_task != exec_wrapper)
118  /* We do not recognize the format! */
119  return -1;
120 
121  if(pinfo->argl > PROCINFO_MAX_ARGS_SIZE)
122  /* The full argument is not available */
123  return -1;
124 
125  int argl = pinfo->argl;
126  void* args = pinfo->args;
127 
128  /* unpack the program pointer */
129  if(prog) memcpy(&prog, args, sizeof(prog));
130 
131  argl -= sizeof(Program*);
132  args += sizeof(Program*);
133 
134  /* unpack the string vector */
135  size_t N = argscount(argl, args);
136  if(argv) {
137  if(argc>N)
138  argc = N;
139  argvunpack(argc, argv, argl, args);
140  }
141 
142  return N;
143 }

147 int Execute(Program prog, size_t argc, const char** argv)
148 {
149  /* We will pack the prog pointer and the arguments to
150  an argument buffer.
151  */
152 
153  /* compute the argument buffer size */
154  size_t argl = argvlen(argc, argv) + sizeof(prog);
155 
156  /* allocate the buffer */
157  char args[argl];
158 
159  /* put the pointer at the start */
160  memcpy(args, &prog, sizeof(prog));
161 
162  /* add the string vector */
163  argvpack(args+sizeof(prog), argc, argv);
164 
165  /* Execute the process */
166  return Exec(exec_wrapper, argl, args);
167 }
