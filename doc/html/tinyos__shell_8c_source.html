139 int RunTerm(size_t argc, const char** argv)
140 {
141  checkargs(2);
142 
143  int term = getint(1);
144  int prog = getprog(2);
145 
146  if(term<0 || term>=GetTerminalDevices()) {
147  printf("The terminal provided is not valid: %d\n", term);
148  return 1;
149  }
150  if(prog<0) {
151  printf("The program provided is not valid: %s\n", argv[2]);
152  return 2;
153  }
154 
155  /* Change our own terminal, so that the child inherits it. */
156  int termfid = OpenTerminal(term);
157  if(termfid!=0) {
158  Dup2(termfid, 0);
159  Close(termfid);
160  }
161  Dup2(0, 1); /* use the same stream for stdout */
162 
163  return Execute(COMMANDS[prog].prog, argc-2, argv+2);
164 }

445 /* the thread that accepts new connections */
446 static int rsrv_listener_thread(int port, void* __globals)
447 {
448  Fid_t lsock = Socket(port);
449  if(Listen(lsock) == -1) {
450  printf("Cannot listen to the given port: %d\n", port);
451  return -1;
452  }
453  GS(listener_socket) = lsock;
454 
455  /* Accept loop */
456  while(1) {
457  Fid_t sock = Accept(lsock);
458  if(sock==NOFILE) {
459  /* We failed! Check if we should quit */
460  if(GS(quit)) return 0;
461  log_message(__globals, "listener(port=%d): failed to accept!\n", port);
462  } else {
463  GS(active_conn)++;
464  GS(total_conn)++;
465  Tid_t t = CreateThread(rsrv_client, sock, __globals);
466  ThreadDetach(t);
467  }
468  }
469  return 0;
470 }

473 /* The main server process */
474 int RemoteServer(size_t argc, const char** argv)
475 {
476  /* Create the globals */
477  struct __rs_globals __global_obj;
478  struct __rs_globals *__globals = &__global_obj;
479 
480  GS(mx) = MUTEX_INIT;
481  GS(conn_done) = COND_INIT;
482 
483  GS(quit) = 0;
484  GS(port) = REMOTE_SERVER_DEFAULT_PORT;
485  GS(active_conn) = 0;
486  GS(total_conn) = 0;
487  GS(conn_id_counter) = 0;
488 
489  log_init(__globals);
490 
491  /* Start a thread to listen on */
492  GS(listener) = CreateThread(rsrv_listener_thread, GS(port), __globals);
493 
494  /* Enter the server console */
495  char* linebuff = NULL;
496  size_t lineblen = 0;
497  FILE* fin = fidopen(0,"r");
498 
499  while(1) {
500  printf("Type h for help, or a command: ");
501  int rc;
502  again:
503  rc = getline(&linebuff, &lineblen, fin);
504  if(rc==-1 && ferror(fin) && errno==EINTR) {
505  clearerr(fin);
506  goto again;
507  }
508  if(rc==-1 && feof(fin)) {
509  break;
510  }
511 
512  assert(linebuff!=NULL);
513 
514  if(strcmp(linebuff, "q\n")==0) {
515  /* Quit */
516  GS(quit) = 1;
517  printf("Quitting\n");
518  Close(GS(listener_socket));
519  ThreadJoin(GS(listener), NULL);
520 
521  Mutex_Lock(&GS(mx));
522  while(GS(active_conn)>0) {
523  printf("Waiting %zu connections ...\n", GS(active_conn));
524  Cond_Wait(&GS(mx), &GS(conn_done));
525  }
526  Mutex_Unlock(&GS(mx));
527 
528 
529  log_truncate(__globals);
530  break;
531  } else if(strcmp(linebuff, "s\n")==0) {
532  /* Show statistics */
533  printf("Connections: active=%4zd total=%4zd\n",
534  GS(active_conn), GS(total_conn));
535  } else if(strcmp(linebuff, "h\n")==0) {
536  printf("Commands: \n"
537  "q: quit the server\n"
538  "h: print this help\n"
539  "s: show statistics\n"
540  "l: show the log\n");
541  } else if(strcmp(linebuff, "l\n")==0) {
542  log_print(__globals);
543  } else if(strcmp(linebuff, "\n")==0) {
544  } else {
545  printf("Unknown command: '%s'\n", linebuff);
546  }
547  }
548 
549  fclose(fin);
550  free(linebuff);
551  return 0;
552 }

559 /* log a message */
560 static void log_message(void* __globals, const char* msg, ...)
561 {
562  /* put the log record in a memory buffer */
563  char* buffer = NULL;
564  size_t buffer_size;
565  FILE* output = open_memstream(&buffer, &buffer_size);
566 
567  /* At the head of the buffer, reserve space for a logrec */
568  fseek(output, sizeof(logrec), SEEK_SET);
569 
570  /* Add the message */
571  va_list ap;
572  va_start (ap, msg);
573  vfprintf (output, msg, ap);
574  va_end (ap);
575  fclose(output);
576 
577  /* Append the record */
578  logrec *rec = (logrec*) buffer;
579  Mutex_Lock(& GS(mx));
580  rlnode_new(& rec->node)->num = ++GS(logcount);
581  rlist_push_back(& GS(log), & rec->node);
582  Mutex_Unlock(& GS(mx));
583 }

636 static int rsrv_process(size_t argc, const char** argv)
637 {
638  checkargs(2);
639  Fid_t sock = atoi(argv[1]);
640 
641  /* Fix the streams */
642  assert(sock!=0 && sock!=1);
643  Dup2(sock, 0);
644  Dup2(sock, 1);
645  Close(sock);
646 
647  /* (a) find the command */
648  int c = getprog(2);
649  if(c==-1) {
650  /* This will appear in the rcli console */
651  printf("Error in remote process: Command %s is not found\n", argv[2]);
652  return -1;
653  }
654  Program proc = COMMANDS[c].prog;
655 
656  /* Execute */
657  int exitstatus;
658  WaitChild(Execute(proc, argc-2, argv+2), &exitstatus);
659  return exitstatus;

662 /* this server thread serves a remote cliend */
663 static int rsrv_client(int sock, void* __globals)
664 {
665  /* Get your client id */
666  Mutex_Lock(&GS(mx));
667  size_t ID = ++GS(conn_id_counter);
668  Mutex_Unlock(&GS(mx));
669 
670  log_message(__globals, "Client[%6zu]: started", ID);
671 
672  /* Get the command from the client. The protocol is
673  [int argl, void* args] where argl is the length of
674  the subsequent message args.
675  */
676  int argl;
677  if(! recv_message(sock, &argl, sizeof(argl))) {
678  log_message(__globals,
679  "Cliend[%6zu]: error in receiving request, aborting", ID);
680  goto finish;
681  }
682 
683  assert(argl>0 && argl <= 2048);
684  {
685  char args[argl];
686  if(! recv_message(sock, args, argl)) {
687  log_message(__globals,
688  "Cliend[%6zu]: error in receiving request, aborting", ID);
689  goto finish;
690  }
691 
692  /* Prepare to execute subprocess */
693  size_t argc = argscount(argl, args);
694  const char* argv[argc+2];
695  argv[0] = "rsrv_process";
696  char sock_value[32];
697  sprintf(sock_value, "%d", sock);
698  argv[1] = sock_value;
699  argvunpack(argc, argv+2, argl, args);
700 
701  /* Now, execute the message in a new process */
702  int exitstatus;
703  Pid_t pid = Execute(rsrv_process, argc+2, argv);
704  Close(sock);
705  WaitChild(pid, &exitstatus);
706 
707  log_message(__globals, "Client[%6zu]: finished with status %d",
708  ID, exitstatus);
709  }
710 
711 finish:
712  Mutex_Lock(&GS(mx));
713  GS(active_conn)--;
714  Cond_Broadcast(& GS(conn_done));
715  Mutex_Unlock(&GS(mx));
716  return 0;
717 }

802 int process_line(int argc, const char** argv)
803 {
804  /* Split up into pipeline fragments */
805  int Vargc[argc];
806  Vargc[0]=0;
807  const char** Vargv[argc];
808 
809  int frag=0;
810 
811  for(int i=0; i<argc; i++)
812  if(strcmp(argv[i],"|")!=0) {
813  Vargc[frag] ++;
814  if(Vargc[frag]==1)
815  Vargv[frag] = argv+i;
816  }
817  else
818  Vargc[++frag] = 0;
819  frag++;
820 
821  /* Check that no fragment is empty and each has a program */
822  int comd[frag];
823  for(int i=0; i<frag; i++) {
824  if(Vargc[i]==0) {
825  printf("Error: a pipeline fragment was empty.\n");
826  return 0;
827  }
828  int c;
829  for(c=0; COMMANDS[c].cmdname; c++)
830  if(strcmp(COMMANDS[c].cmdname, Vargv[i][0])==0)
831  break;
832  if(! COMMANDS[c].cmdname) {
833  printf("Command not found: '%s'\n", Vargv[i][0]);
834  return 0;
835  }
836  comd[i] = c;
837  }
838 
839  /* Construct pipeline */
840  int child[frag];
841  int savein, saveout;
842 
843  savein = savefid(0);
844  saveout = savefid(1);
845 
846  pipe_t pipe;
847  for(int i=0; i<frag; i++) {
848  if(i<frag-1) {
849  /* Not the last fragment, make a pipe */
850  Pipe(& pipe);
851  Dup2(pipe.write,1);
852  Close(pipe.write);
853  } else {
854  /* Last fragment, restore saved 1 */
855  Dup2(saveout, 1);
856  Close(saveout);
857  }
858 
859  child[i] = Execute(COMMANDS[comd[i]].prog, Vargc[i], Vargv[i]);
860 
861  if(i<frag-1) {
862  /* Not the last fragment, make a pipe */
863  Dup2(pipe.read,0);
864  Close(pipe.read);
865  } else {
866  /* Last fragment, restore saved 1 */
867  Dup2(savein, 0);
868  Close(savein);
869  }
870  }
871 
872  /* Wait for the children */
873  for(int i=0; i<frag; i++) {
874  int exitval;
875  WaitChild(child[i], &exitval);
876  if(exitval)
877  printf("%s exited with status %d\n", Vargv[i][0], exitval);
878  }
879 
880  return 1;
881 }
