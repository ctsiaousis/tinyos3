83 void serial_rx_handler()
84 {
85  int pre = preempt_off;
86 
87  /*
88  We do not know which terminal is
89  ready, so we must signal them all !
90  */
91  for(int i=0;i<bios_serial_ports();i++) {
92  serial_dcb_t* dcb = &serial_dcb[i];
93  Cond_Broadcast(&dcb->rx_ready);
94  }
95  if(pre) preempt_on;
96 }

98 /*
99  Read from the device, sleeping if needed.
100  */
101 int serial_read(void* dev, char *buf, unsigned int size)
102 {
103  serial_dcb_t* dcb = (serial_dcb_t*)dev;
104 
105  preempt_off; /* Stop preemption */
106 
107  uint count = 0;
108 
109  while(count<size) {
110  int valid = bios_read_serial(dcb->devno, &buf[count]);
111 
112  if (valid) {
113  count++;
114  }
115  else if(count==0) {
116  kernel_wait(&dcb->rx_ready, SCHED_IO);
117  }
118  else
119  break;
120  }
121 
122  preempt_on; /* Restart preemption */
123 
124  return count;
125 }
