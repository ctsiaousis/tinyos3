249 static Pid_t wait_for_specific_child(Pid_t cpid, int* status)
250 {
251 
252  /* Legality checks */
253  if((cpid<0) || (cpid>=MAX_PROC)) {
254  cpid = NOPROC;
255  goto finish;
256  }
257 
258  PCB* parent = CURPROC;
259  PCB* child = get_pcb(cpid);
260  if( child == NULL || child->parent != parent)
261  {
262  cpid = NOPROC;
263  goto finish;
264  }
265 
266  /* Ok, child is a legal child of mine. Wait for it to exit. */
267  while(child->pstate == ALIVE)
268  kernel_wait(& parent->child_exit, SCHED_USER);
269 
270  cleanup_zombie(child, status);
271 
272 finish:
273  return cpid;
274 }

277 static Pid_t wait_for_any_child(int* status)
278 {
279  Pid_t cpid;
280 
281  PCB* parent = CURPROC;
282 
283  /* Make sure I have children! */
284  if(is_rlist_empty(& parent->children_list)) {
285  cpid = NOPROC;
286  goto finish;
287  }
288 
289  while(is_rlist_empty(& parent->exited_list)) {
290  kernel_wait(& parent->child_exit, SCHED_USER);
291  }
292 
293  PCB* child = parent->exited_list.next->pcb;
294  assert(child->pstate == ZOMBIE);
295  cpid = get_pid(child);
296  cleanup_zombie(child, status);
297 
298 finish:

317 void sys_Exit(int exitval)
318 {
319  /* Right here, we must check that we are not the boot task. If we are,
320  we must wait until all processes exit. */
321  if(sys_GetPid()==1) {
322  while(sys_WaitChild(NOPROC,NULL)!=NOPROC);
323  }
324 
325  PCB *curproc = CURPROC; /* cache for efficiency */
326 
327  /* Do all the other cleanup we want here, close files etc. */
328  if(curproc->args) {
329  free(curproc->args);
330  curproc->args = NULL;
331  }
332 
333  /* Clean up FIDT */
334  for(int i=0;i<MAX_FILEID;i++) {
335  if(curproc->FIDT[i] != NULL) {
336  FCB_decref(curproc->FIDT[i]);
337  curproc->FIDT[i] = NULL;
338  }
339  }
340 
341  /* Reparent any children of the exiting process to the
342  initial task */
343  PCB* initpcb = get_pcb(1);
344  while(!is_rlist_empty(& curproc->children_list)) {
345  rlnode* child = rlist_pop_front(& curproc->children_list);
346  child->pcb->parent = initpcb;
347  rlist_push_front(& initpcb->children_list, child);
348  }
349 
350  /* Add exited children to the initial task's exited list
351  and signal the initial task */
352  if(!is_rlist_empty(& curproc->exited_list)) {
353  rlist_append(& initpcb->exited_list, &curproc->exited_list);
354  kernel_broadcast(& initpcb->child_exit);
355  }
356 
357  /* Put me into my parent's exited list */
358  if(curproc->parent != NULL) { /* Maybe this is init */
359  rlist_push_front(& curproc->parent->exited_list, &curproc->exited_node);
360  kernel_broadcast(& curproc->parent->child_exit);
361  }
362 
363  /* Disconnect my main_thread */
364  curproc->main_thread = NULL;
365 
366  /* Now, mark the process as exited. */
367  curproc->pstate = ZOMBIE;
368  curproc->exitval = exitval;
369 
370  /* Bye-bye cruel world */
371  kernel_sleep(EXITED, SCHED_USER);
372 }
