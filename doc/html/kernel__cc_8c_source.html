22 /*
23  Pre-emption aware mutex.
24  -------------------------
25 
26  This mutex will act as a spinlock if preemption is off, and a
27  yielding mutex if preemption is on.
28 
29  Therefore, we can call the same function from both the preemptive and
30  the non-preemptive domain of the kernel.
31 
32  The implementation is based on GCC atomics, as the standard C11 primitives
33  are not supported by all recent compilers. Eventually, this will change.
34  */
35 void Mutex_Lock(Mutex* lock)
36 {
37 #define MUTEX_SPINS 1000
38 
39  while(__atomic_test_and_set(lock,__ATOMIC_ACQUIRE)) {
40  int spin=MUTEX_SPINS;
41  while(__atomic_load_n(lock, __ATOMIC_RELAXED)) {
42  __builtin_ia32_pause();
43  if(spin>0)
44  spin--;
45  else {
46  spin=MUTEX_SPINS;
47  if(get_core_preemption())
48  yield(SCHED_MUTEX);
49  }
50  }
51  }
52 #undef MUTEX_SPINS
53 }

119 static int cv_wait(Mutex* mutex, CondVar* cv,
120  enum SCHED_CAUSE cause, TimerDuration timeout)
121 {
122  __cv_waiter waiter = { .thread=CURTHREAD, .signalled = 0, .removed=0 };
123  rlnode_init(& waiter.node, &waiter);
124 
125  Mutex_Lock(&(cv->waitset_lock));
126  /* We just push the current thread to the back of the list */
127  if(cv->waitset) {
128  __cv_waiter* wset = cv->waitset;
129  rlist_push_back(& wset->node, & waiter.node);
130  } else {
131  cv->waitset = &waiter;
132  }
133 
134  /* Now atomically release mutex and sleep */
135  Mutex_Unlock(mutex);
136  sleep_releasing(STOPPED, &(cv->waitset_lock), cause, timeout);
137 
138  /* Woke up, we must check wether we were signaled, and tidy up */
139  Mutex_Lock(&(cv->waitset_lock));
140  if(! waiter.removed) {
141  assert(! waiter.signalled);
142 
143  /* We must remove ourselves from the ring! */
144  remove_from_ring(cv, &waiter);
145  }
146  Mutex_Unlock(&(cv->waitset_lock));
147 
148  Mutex_Lock(mutex);
149  return waiter.signalled;
150 }

159 static inline void cv_signal(CondVar* cv)
160 {
161  /* Wakeup first process in the waiters' queue, if it exists. */
162  while(cv->waitset) {
163  __cv_waiter* waiter = cv->waitset;
164  remove_from_ring(cv, waiter);
165  waiter->removed = 1;
166  if(wakeup(waiter->thread)) {
167  waiter->signalled = 1;
168  return;
169  }
170  }
